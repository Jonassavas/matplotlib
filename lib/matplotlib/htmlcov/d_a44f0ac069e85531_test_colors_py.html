1104def test_light_source_hillshading(): 

1105 """ 

1106 Compare the current hillshading method against one that should be 

1107 mathematically equivalent. Illuminates a cone from a range of angles. 

1108 """ 

1109 

1110 def alternative_hillshade(azimuth, elev, z): 

1111 illum = _sph2cart(*_azimuth2math(azimuth, elev)) 

1112 illum = np.array(illum) 

1113 

1114 dy, dx = np.gradient(-z) 

1115 dy = -dy 

1116 dz = np.ones_like(dy) 

1117 normals = np.dstack([dx, dy, dz]) 

1118 normals /= np.linalg.norm(normals, axis=2)[..., None] 

1119 

1120 intensity = np.tensordot(normals, illum, axes=(2, 0)) 

1121 intensity -= intensity.min() 

1122 intensity /= intensity.ptp() 

1123 return intensity 

1124 

1125 y, x = np.mgrid[5:0:-1, :5] 

1126 z = -np.hypot(x - x.mean(), y - y.mean())

1129 ls = mcolors.LightSource(az, elev)

1132 assert_array_almost_equal(h1, h2)

1135def test_light_source_planar_hillshading(): 

1136 """ 

1137 Ensure that the illumination intensity is correct for planar surfaces. 

1138 """ 

1139 

1140 def plane(azimuth, elevation, x, y): 

1141 """ 

1142 Create a plane whose normal vector is at the given azimuth and 

1143 elevation. 

1144 """ 

1145 theta, phi = _azimuth2math(azimuth, elevation) 

1146 a, b, c = _sph2cart(theta, phi) 

1147 z = -(a*x + b*y) / c 

1148 return z

1150 def angled_plane(azimuth, elevation, angle, x, y): 

1151 """ 

1152 Create a plane whose normal vector is at an angle from the given 

1153 azimuth and elevation. 

1154 """ 

1155 elevation = elevation + angle 

1156 if elevation > 90: 

1157 azimuth = (azimuth + 180) % 360 

1158 elevation = (90 - elevation) % 90 

1159 return plane(azimuth, elevation, x, y) 

1160 

1161 y, x = np.mgrid[5:0:-1, :5] 

1162 for az, elev in itertools.product(range(0, 390, 30), range(0, 105, 15)): 

1163 ls = mcolors.LightSource(az, elev) 

1164 

1165 # Make a plane at a range of angles to the illumination 

1166 for angle in range(0, 105, 15): 

1167 z = angled_plane(az, elev, angle, x, y) 

1168 h = ls.hillshade(z)
